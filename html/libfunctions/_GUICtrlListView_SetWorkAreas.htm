Function Reference

_GUICtrlListView_SetWorkAreas

コントロール内の作業領域を作成します。

#Include <GuiListView.au3>
_GUICtrlListView_SetWorkAreas($hWnd, $iLeft, $iTop, $iRight, $iBottom)

 

パラメータ

$hWnd コントロールのハンドル
$iLeft 長方形の左上隅のX座標
$iTop 長方形の左上隅のY座標
$iRight 長方形の右下隅のX座標
$iBottom 長方形の右下隅のY座標

 

返し値

なし。

 

注意

なし。

 

関連

_GUICtrlListView_GetNumberOfWorkAreas

 

例


#AutoIt3Wrapper_au3check_parameters=-d -w 1 -w 2 -w 3 -w 4 -w 5 -w 6
#include <GuiConstantsEx.au3>
#include <GuiListView.au3>
#include <GuiImageList.au3>

Opt('MustDeclareVars', 1)

$Debug_LV = False ; ListView関数に渡されるClassNameを調べる。動作を確認するにはTrueを設定し、他のコントロールのハンドルを使用

_Main()

Func _Main()
    Local $hImage, $hListView
   
    GUICreate("ListView Set Work Areas", 400, 300)
    $hListView = GUICtrlCreateListView("", 2, 2, 394, 268)
    GUICtrlSetStyle($hListView, $LVS_ICON)
    GUISetState()

    ; 画像をロード
    $hImage = _GUIImageList_Create()
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($hListView, 0xFF0000, 16, 16))
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($hListView, 0x00FF00, 16, 16))
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($hListView, 0x0000FF, 16, 16))
    _GUICtrlListView_SetImageList($hListView, $hImage, 0)

    ; 列を追加
    _GUICtrlListView_AddColumn($hListView, 0, "Items", 100)

    ; アイテムを追加
    _GUICtrlListView_AddItem($hListView, "Item 1", 0)
    _GUICtrlListView_AddItem($hListView, "Item 2", 1)
    _GUICtrlListView_AddItem($hListView, "Item 3", 2)

    ; 作業領域を作成
    _GUICtrlListView_SetWorkAreas($hListView, 0, 0, 100, 100)
    MsgBox(4160, "Information", "Work Areas: " & _GUICtrlListView_GetNumberOfWorkAreas($hListView))

    ; ユーザーが終了するまでループ
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>_Main
