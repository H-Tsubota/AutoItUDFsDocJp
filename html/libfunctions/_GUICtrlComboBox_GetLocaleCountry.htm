_GUICtrlComboBox_GetLocaleCountry

現在の国コードを取得します。

#Include <GuiComboBox.au3>
_GUICtrlComboBox_GetLocaleCountry($hWnd)

 

パラメータ

$hWnd コントロールのハンドル

 

返し値

成功: コンボボックスの国コード

 

注意

なし。

 

関連

_GUICtrlComboBox_GetLocale

 

例


#AutoIt3Wrapper_au3check_parameters=-d -w 1 -w 2 -w 3 -w 4 -w 5 -w 6
#include <GUIComboBox.au3>
#include <GuiConstantsEx.au3>

Opt('MustDeclareVars', 1)

$Debug_CB = False ; ComboBox/ComboBoxEx関数に渡されるClassNameを調べる。動作を確認するにはTrueを設定し、他のコントロールのハンドルを使用

_Main()

Func _Main()
    Local $hCombo

    ; GUIを作成
    GUICreate("ComboBox Get Locale Country Code", 400, 296)
    $hCombo = GUICtrlCreateCombo("", 2, 2, 396, 296)
    GUISetState()

    ; ファイルを追加
    _GUICtrlComboBox_BeginUpdate($hCombo)
    _GUICtrlComboBox_AddDir($hCombo, @WindowsDir & "\*.exe")
    _GUICtrlComboBox_EndUpdate($hCombo)

    ; ロケール、国コード、言語識別子、プライマリ言語ID、サブ言語IDを表示
    MsgBox(4160, "Information", _
            "Locale .................: " & _GUICtrlComboBox_GetLocale($hCombo) & @LF & _
            "Country code ........: " & _GUICtrlComboBox_GetLocaleCountry($hCombo) & @LF & _
            "Language identifier..: " & _GUICtrlComboBox_GetLocaleLang($hCombo) & @LF & _
            "Primary Language id : " & _GUICtrlComboBox_GetLocalePrimLang($hCombo) & @LF & _
            "Sub-Language id ....: " & _GUICtrlComboBox_GetLocaleSubLang($hCombo))

    ; ユーザーが終了するまでループ
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>_Main
