Function Reference

_GUICtrlComboBox_GetComboBoxInfo

指定されたコンボボックスの情報を取得します。

#Include <GuiComboBox.au3>
_GUICtrlComboBox_GetComboBoxInfo($hWnd, ByRef $tInfo)

 

パラメータ

$hWnd コントロールのハンドル
$tInfo コンボボックスの情報を受け取る変数

 

返し値

成功: True
失敗: False

 

注意

必須OS: Windows XP以降

 

関連

 

例


#AutoIt3Wrapper_au3check_parameters=-d -w 1 -w 2 -w 3 -w 4 -w 5 -w 6
#include <GUIComboBox.au3>
#include <GuiConstantsEx.au3>
#include <Constants.au3>

Opt('MustDeclareVars', 1)

$Debug_CB = False ; ComboBox/ComboBoxEx関数に渡されるClassNameを調べる。動作を確認するにはTrueを設定し、他のコントロールのハンドルを使用

Global $iMemo

_Main()

Func _Main()
    Local $tInfo, $hCombo

    ; GUIを作成
    GUICreate("ComboBox Get ComboBox Info", 400, 296)
    $hCombo = GUICtrlCreateCombo("", 2, 2, 396, 296)
    $iMemo = GUICtrlCreateEdit("", 2, 32, 396, 266, 0)
    GUICtrlSetFont($iMemo, 9, 400, 0, "Courier New")
    GUISetState()

    ; ファイルを追加
    _GUICtrlComboBox_BeginUpdate($hCombo)
    _GUICtrlComboBox_AddDir($hCombo, @WindowsDir & "\*.exe")
    _GUICtrlComboBox_EndUpdate($hCombo)

    If _GUICtrlComboBox_GetComboBoxInfo($hCombo, $tInfo) Then
        MemoWrite("Handle to the ComboBox .....: " & DllStructGetData($tInfo, "hCombo"))
        MemoWrite("Handle to the Edit Box .....: " & DllStructGetData($tInfo, "hEdit"))
        MemoWrite("Handle to the drop-down list: " & DllStructGetData($tInfo, "hList"))
    EndIf
   
    ; ユーザーが終了するまでループ
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>_Main

; メモコントロールに1行書き込み
Func MemoWrite($sMessage)
    GUICtrlSetData($iMemo, $sMessage & @CRLF, 1)
EndFunc   ;==>MemoWrite
