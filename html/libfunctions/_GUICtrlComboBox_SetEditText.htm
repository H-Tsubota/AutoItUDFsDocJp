Function Reference

_GUICtrlComboBox_SetEditText

コンボボックスのエディットコントロールのテキストを設定します。

#Include <GuiComboBox.au3>
_GUICtrlComboBox_SetEditText($hWnd, $sText)

 

パラメータ

$hWnd コントロールのハンドル
$sText 設定するテキスト

 

返し値

なし。

 

注意

必須OS: Windows XP以降

$CBS_DROPDOWNスタイルまたは$CBS_DROPDOWNLISTスタイルが設定されたコンボボックスに使用した場合、この関数は失敗します。

 

関連

_GUICtrlComboBox_GetEditText, _GUICtrlComboBox_ReplaceEditSel

 

例


#AutoIt3Wrapper_au3check_parameters=-d -w 1 -w 2 -w 3 -w 4 -w 5 -w 6
#include <GUIComboBox.au3>
#include <GuiConstantsEx.au3>

Opt('MustDeclareVars', 1)

$Debug_CB = False ; ComboBox/ComboBoxEx関数に渡されるClassNameを調べる。動作を確認するにはTrueを設定し、他のコントロールのハンドルを使用

Global $iMemo

_Main()

Func _Main()
    Local $hCombo

    ; GUIを作成
    GUICreate("ComboBox Set Edit Text", 400, 296)
    $hCombo = GUICtrlCreateCombo("", 2, 2, 396, 296)
    $iMemo = GUICtrlCreateEdit("", 2, 32, 396, 266, 0)
    GUICtrlSetFont($iMemo, 9, 400, 0, "Courier New")
    GUISetState()

    ; ファイルを追加
    _GUICtrlComboBox_BeginUpdate($hCombo)
    _GUICtrlComboBox_AddDir($hCombo, @WindowsDir & "\*.exe")
    _GUICtrlComboBox_EndUpdate($hCombo)

    ; エディットのテキストを設定
    _GUICtrlComboBox_SetEditText($hCombo, StringFormat("%d - Some New Text", Random(1, 20, 1)))
   
    ; エディットのテキストを取得
    MemoWrite("Edit Text: " & _GUICtrlComboBox_GetEditText($hCombo))

    ; ユーザーが終了するまでループ
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>_Main

; メモコントロールに1行書き込み
Func MemoWrite($sMessage)
    GUICtrlSetData($iMemo, $sMessage & @CRLF, 1)
EndFunc   ;==>MemoWrite
