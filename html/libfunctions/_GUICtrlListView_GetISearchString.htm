_GUICtrlListView_GetISearchString

コントロールのインクリメント検索文字列を取得します。

#Include <GuiListView.au3>
_GUICtrlListView_GetISearchString($hWnd)

 

パラメータ

$hWnd コントロールのハンドル

 

返し値

インクリメント検索文字列を返します。検索文字列がない場合は空文字列が返されます。

 

注意

インクリメント検索文字列とはリストビューにフォーカスしている際にユーザーが入力した文字の並びです。
ユーザーが文字を入力するたびにシステムは文字を検索文字列に追加し、一致するアイテムを検索します。
システムが一致するものを見つけた場合はそのアイテムが選択され、必要な場合は表示されるようにスクロールされます。

タイムアウト時間はユーザーが入力した各文字に対して関連付けられます。
ユーザーが次の文字を入力する前にタイムアウト時間が経過するとインクリメント検索文字列はリセットされます。

$Debug_LV = False ; ListView関数に渡されるClassNameを調べる。動作を確認するにはTrueを設定し、他のコントロールのハンドルを使用

_Main()

Func _Main()
    Local $hImage, $hListView
   
    GUICreate("ListView Get ISearch", 400, 300)

    $hListView = GUICtrlCreateListView("", 2, 2, 394, 268)
    GUICtrlSetStyle($hListView, $LVS_ICON)
    GUISetState()

    ; 画像をロード
    $hImage = _GUIImageList_Create()
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($hListView, 0xFF0000, 16, 16))
    _GUICtrlListView_SetImageList($hListView, $hImage, 0)
   
    _GUICtrlListView_BeginUpdate($hListView)
    For $x = 1 To 10
        _GUICtrlListView_InsertItem($hListView, "Item " & $x, -1, 0)
    Next
    _GUICtrlListView_EndUpdate($hListView)
   
    Send("Item 1")
   
    ; インクリメント検索文字列を取得
    MsgBox(4160, "Information", "Incremental Search String: " & _GUICtrlListView_GetISearchString($hListView))

    ; ユーザーが終了するまでループ
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    GUIDelete()
EndFunc   ;==>_Main
