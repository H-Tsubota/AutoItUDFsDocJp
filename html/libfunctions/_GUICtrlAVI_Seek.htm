Function Reference

_GUICtrlAVI_Seek

AVIコントロールにAVIクリップの特定のフレームを表示するように指示します。

#Include <GuiAVI.au3>
_GUICtrlAVI_Seek($hWnd, $iFrame)

 

パラメータ

$hWnd コントロールのハンドル
$iFrame 表示するフレームのゼロ始まりのインデックス

 

返し値

成功: True
失敗: False

 

注意

なし。

 

関連

_GUICtrlAVI_Play

 

例


#include <GuiConstantsEx.au3>
#include <GuiAVI.au3>

Opt('MustDeclareVars', 1)

$Debug_AVI = False ; AVI関数に渡されるClassNameを調べる。動作を確認するにはTrueを設定し、他のコントロールのハンドルを使用

Global $hAVI

_Example_Internal()
_Example_External()

Func _Example_Internal()
    Local $hGUI

    ; GUIを作成
    $hGUI = GUICreate("(Internal) AVI Seek", 300, 100)
    $hAVI = GUICtrlCreateAvi(@SystemDir & "\shell32.dll", 160, 10, 10)
    GUISetState()

    ; ユーザーが終了するまでループ
    Do
        Sleep(100)
        ; AVIクリップのランダムなフレームへシーク
        _GUICtrlAVI_Seek ($hAVI, Random(1, 30, 1))
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
   
    ; AVIクリップを閉じる
    _GUICtrlAVI_Close ($hAVI)

   
    GUIDelete()
EndFunc   ;==>_Main

Func _Example_External()
    Local $hGUI

    ; GUIを作成
    $hGUI = GUICreate("(External) AVI Seek", 300, 100)
    $hAVI = _GUICtrlAVI_Create ($hGUI, @SystemDir & "\Shell32.dll", 160, 10, 10)
    GUISetState()

    ; ユーザーが終了するまでループ
    Do
        Sleep(100)
        ; AVIクリップのランダムなフレームへシーク
        _GUICtrlAVI_Seek ($hAVI, Random(1, 30, 1))
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
   
    ; AVIクリップを閉じる
    _GUICtrlAVI_Close ($hAVI)

   
    GUIDelete()
EndFunc   ;==>_Main
